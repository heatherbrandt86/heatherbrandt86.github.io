Bolt Era: Is Nebraska back?

2021-04-25 
library(tidyverse)
library(ggalt)
teamhitting <- read_csv("Baseball Hitting Stats.csv")
teampitching <- read_csv("Baseball Pitching Stats.csv")
runsscored <- teamhitting %>% rename(RunsScored = R) %>% select(Season, School, RunsScored) 
runsallowed <- teampitching %>% select(Season, School, R) %>% rename(RunsAllowed = R)

runs <- runsscored %>% full_join(runsallowed) %>% mutate(SchoolSeason = paste(School, Season, sep=" "))
## Joining, by = c("Season", "School")
current <- runs %>% filter(Season == "2021")
nu <- runs %>% filter(School == "Nebraska" & Season == "2021")
ggplot() + geom_point(data=current, aes(x=RunsScored, y=RunsAllowed)) +
  geom_point(data=nu, aes(x=RunsScored, y=RunsAllowed), color="red")

thisseason<- teampitching %>% filter(Season == "2021", Conference == "Big Ten")
ggplot() + 
  geom_dumbbell(
    data=thisseason, 
    aes(y=reorder(School,SO), x=SO, xend=BB),
    size = 2,
colour = "light gray",
    colour_x = "black",
    colour_xend = "red")+
  theme_minimal()+
  labs(y="Strikouts", x= "Team", 
       title = "Husker's Pitching Showing Promise", 
subtitle ="")

lastsixseasons<- teamhitting %>% filter(School == "Nebraska")


ggplot() + 
  geom_line(data=lastsixseasons, aes(x=Season, y=OBPct))
